MINVALUE 1
START WITH 1
INCREMENT BY 1
CACHE 20;

idReteta NUMBER(6) PRIMARY KEY,

);

MINVALUE 1
START WITH 1
INCREMENT BY 1
CACHE 20;

CREATE TABLE INGREDIENT_31A_FPC(
idIngredient NUMBER(4) PRIMARY KEY,
numeIngredient VARCHAR2(50) NOT NULL,
dataExp DATE DEFAULT CURRENT_DATE,
stoc NUMBER(6)
);

MINVALUE 1

idMedicament NUMBER(6) PRIMARY KEY,
